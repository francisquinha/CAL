164  {
165  this->maxCost=maxCost;

169  BnB_elementDATA elem_data = {
170  ((*it).*getCost)(), // cost

174  if (it==elements.begin() || elem_data.cost<minItemCost) minItemCost = elem_data.cost;
175  elem_data.value_cost_ratio = elem_data.value/elem_data.cost;
176  ordered_elements.push_back( pair<BnB_elementDATA,T*>(
177  elem_data
178  , &(*it) ) );
179  }
